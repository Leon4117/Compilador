int i;
int j;
float v;
float x;
float[10] a;

// Initialize array
print "Initializing array...";
i = 0;
while (i < 10) {
    a[i] = 10 - i; // 10, 9, 8, ...
    print a[i];
    i = i + 1;
}

v = 5; // Pivot
i = -1;
j = 10;

print "Start partitioning...";

while (true) {
    do {
        i = i + 1;
    } while (a[i] < v);

    do {
        j = j - 1;
    } while (a[j] > v);

    if (i >= j) break;

    print "Swapping elements at indices:";
    print i;
    print j;

    x = a[i];
    a[i] = a[j];
    a[j] = x;
}

print "Partitioning complete.";
print "Array after partitioning:";

i = 0;
while (i < 10) {
    print a[i];
    i = i + 1;
}
