var a = 10;
var b = 20;
print "Sum:";
print a + b;

if (a < b) {
    print "a is smaller";

print "While loop:";
var i = 0;
while (i < 3) {
    print i;
    i = i + 1;
}

print "For loop with continue/break:";
for (var j = 0; j < 5; j = j + 1) {
    if (j == 2) continue;
    if (j == 4) break;
    print j;
}
