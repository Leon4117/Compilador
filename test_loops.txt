// Test 3: While loops
print "Testing while loops:";

// Simple while loop
var counter = 0;
print "Counting from 0 to 4:";
while (counter < 5) {
    print counter;
    counter = counter + 1;
}

// While loop with condition
var sum = 0;
var i = 1;
print "Sum of numbers 1 to 5:";
while (i <= 5) {
    sum = sum + i;
    i = i + 1;
}
print "Sum = " + sum;

// Nested while loops
print "Multiplication table (1-3):";
var row = 1;
while (row <= 3) {
    var col = 1;
    while (col <= 3) {
        var product = row * col;
        print row + " x " + col + " = " + product;
        col = col + 1;
    }
    row = row + 1;
}
